class SVM:

  def __init__(self, learningRate=0.001, iterations=1000, lambda_param=0.01):
    self.learningRate = learningRate
    self.iterations = iterations
    self.lambda_param=lambda_param
    self.w = None
    self.b = None

  def fit(self, X, Y):
    N, D = X.shape
    self.w = np.zeros(D)
    self.b = 0

    for _ in range(self.iterations):
      for index, x in enumerate(X):

        if Y[index] * (self.w @ x + self.b) >= 1:
          self.w -= self.learningRate*2*self.lambda_param*self.w
        else:
          self.w -= self.learningRate * 2 * self.lambda_param * self.w - Y[index]*x
          self.b -= -self.learningRate * Y[index]
